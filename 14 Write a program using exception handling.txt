#include <iostream>

using namespace std;

class User{
private:
    int age;
    float salary;
    string city;
    int haveFourWeeler;
public:
    User(int age,float salary,string city,int haveFourWeeler);
};

User::User(int age,float salary,string city,int haveFourWeeler){
    try{
      if(age<18 || age > 55)
        throw 1;
      if(salary<50000 || salary > 100000)
        throw 2;
      if(city != "pune" && city != "mumbai" && city != "chennai" && city != "bangalore")
        throw 3;
      if(!haveFourWeeler)
        throw 4;

        this->age = age;
        this->city = city;
        this->haveFourWeeler = haveFourWeeler;
        this->salary = salary;
    }
    catch(int e){
        cout<<"Error "<<e<<endl;
    }
}

int main()
{
    User u(3,55000,"pune",1);
    return 0;
}
